#include <unordered_map>
#include <string>
#include <vector>
#include <algorithm>
#include <cctype>

// 自定义哈希函数
struct CaseInsensitiveHash {
    std::size_t operator()(const std::string &s) const {
        std::string lower_s = s;
        std::transform(lower_s.begin(), lower_s.end(), lower_s.begin(),
                       [](unsigned char c) { return std::tolower(c); });
        return std::hash<std::string>{}(lower_s);
    }
};

// 自定义比较函数
struct CaseInsensitiveEqual {
    bool operator()(const std::string &lhs, const std::string &rhs) const {
        return lhs.size() == rhs.size() &&  
               std::equal(lhs.begin(), lhs.end(), rhs.begin(),
                          [](unsigned char a, unsigned char b) {
                              return std::tolower(a) == std::tolower(b);
                          });
    }
};

// field 类定义
class field {
public:
    using comment_container = std::vector<std::string>;
};

// 先声明 basic_inifile
template <typename Hash, typename Equal>
class basic_inifile;

// ini section（支持大小写不敏感键）
template <typename Hash, typename Equal>
class basic_section {
    // 允许 basic_inifile 访问 private 成员
    template <typename H, typename E>
    friend class basic_inifile;

    using DataContainer = std::unordered_map<std::string, field, Hash, Equal>;

public:
    using comment_container = field::comment_container;

    using key_type = typename DataContainer::key_type;
    using mapped_type = typename DataContainer::mapped_type;
    using value_type = typename DataContainer::value_type;
    using size_type = typename DataContainer::size_type;
    using difference_type = typename DataContainer::difference_type;

    using iterator = typename DataContainer::iterator;
    using const_iterator = typename DataContainer::const_iterator;

    // 默认构造
    basic_section() = default;
    // 默认析构函数
    ~basic_section() = default;
    // 默认移动构造函数
    basic_section(basic_section &&) noexcept = default;
    // 默认移动赋值函数
    basic_section &operator=(basic_section &&) noexcept = default;

private:
    DataContainer data_; // 存储键值对
};

// ini 文件类，存储多个 section
template <typename Hash, typename Equal>
class basic_inifile {
public:
    using section = basic_section<Hash, Equal>;  // 在 basic_inifile 内部定义 section 别名
    using SectionContainer = std::unordered_map<std::string, section, Hash, Equal>;

    using key_type = typename SectionContainer::key_type;
    using mapped_type = typename SectionContainer::mapped_type;
    using value_type = typename SectionContainer::value_type;
    using size_type = typename SectionContainer::size_type;
    using difference_type = typename SectionContainer::difference_type;

    using iterator = typename SectionContainer::iterator;
    using const_iterator = typename SectionContainer::const_iterator;

    // 默认构造
    basic_inifile() = default;
    // 默认析构
    ~basic_inifile() = default;
    // 默认移动构造
    basic_inifile(basic_inifile &&) noexcept = default;
    // 默认移动赋值
    basic_inifile &operator=(basic_inifile &&) noexcept = default;

    // 添加一个 section
    section &add_section(const std::string &name) {
        return sections_[name]; // 如果 name 存在，返回已有的 section，否则创建新 section
    }

    // 获取一个 section（可变版本）
    section *get_section(const std::string &name) {
        auto it = sections_.find(name);
        return it != sections_.end() ? &it->second : nullptr;
    }

    // 获取一个 section（只读版本）
    const section *get_section(const std::string &name) const {
        auto it = sections_.find(name);
        return it != sections_.end() ? &it->second : nullptr;
    }

private:
    SectionContainer sections_; // 存储多个 section
};

// 提供默认大小写不敏感的类型别名（类似 std::string）
using section = basic_section<CaseInsensitiveHash, CaseInsensitiveEqual>;
using inifile = basic_inifile<CaseInsensitiveHash, CaseInsensitiveEqual>;

#include <iostream>

int main() {
    inifile ini;  // 默认使用大小写不敏感模式

    ini.add_section("General");
    ini.add_section("Database");

    // 测试大小写不敏感
    if (ini.get_section("general")) {
        std::cout << "Section 'General' exists (case insensitive)" << std::endl;
    }

    if (ini.get_section("DATABASE")) {
        std::cout << "Section 'Database' exists (case insensitive)" << std::endl;
    }

    return 0;
}
