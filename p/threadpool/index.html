{
  void member_function(int y)

    std::cout << "MyClass::member_function: " << y << std::endl;
  }
  int add(int a, int b)
  {
    return a + b;
  }

  void operator()(const std::string& msg) const
  {
    std::cout << "Functor called with: " << msg << std::endl;
