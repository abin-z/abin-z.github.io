  /// @brief ÂΩìÂâçÁ∫øÁ®ãÊ±†ÁöÑÊÄªÁ∫øÁ®ãÊï∞Èáè
  std::size_t total_threads() const noexcept
  {
    return workers_.size();
  }
  /// @brief Ëé∑ÂèñÂΩìÂâçÁ≠âÂæÖÁöÑ‰ªªÂä°Êï∞Èáè
  std::size_t pending_tasks() const noexcept
  {
    std::lock_guard<std::mutex> lock(mtx_);
    return task_queue_.size();
  }
  /// @brief Ëé∑ÂèñÁπÅÂøôÁöÑÁ∫øÁ®ãÊï∞Èáè
  std::size_t busy_threads() const noexcept
  {
    return busy_count_.load();
  }
  /// @brief Ëé∑ÂèñÁ©∫Èó≤Á∫øÁ®ãÊï∞Èáè
  std::size_t idle_threads() const noexcept
  {
    return workers_.size() - busy_count_.load();
  }
  /// @brief ÂΩìÂâçÁ∫øÁ®ãÊ±†ÊòØÂê¶Ê≠£Âú®ËøêË°å(Êú™ÂÅúÊ≠¢)
  bool is_running() const noexcept
  {
    return running_.load();
  }

  /// @brief Ëé∑ÂèñÁ∫øÁ®ãÊ±†ÁöÑÂΩìÂâçÁä∂ÊÄÅ‰ø°ÊÅØ
  status_info status() const noexcept
  {
    std::size_t total = 0;
    std::size_t pending = 0;
    {
      std::lock_guard<std::mutex> lock(mtx_);
      total = workers_.size();
      pending = task_queue_.size();
    }
    std::size_t busy = busy_count_.load();
    std::size_t idle = total - busy;
    return {total, busy, idle, pending, running_.load()};
  }

  // Á¶ÅÁî®Êã∑Ë¥ùÊûÑÈÄ†ÂáΩÊï∞ÂíåÊã∑Ë¥ùËµãÂÄºÊìç‰ΩúÁ¨¶
  threadpool(const threadpool &) = delete;
  threadpool &operator=(const threadpool &) = delete;
  // Á¶ÅÁî®ÁßªÂä®ÊûÑÈÄ†ÂáΩÊï∞ÂíåÁßªÂä®ËµãÂÄºÊìç‰ΩúÁ¨¶
  threadpool(threadpool &&) = delete;
  threadpool &operator=(threadpool &&) = delete;

 private:
  /// @brief ÈªòËÆ§Á∫øÁ®ãÊï∞, Ëé∑ÂèñÁ°¨‰ª∂ÊîØÊåÅÁöÑÂπ∂ÂèëÁ∫øÁ®ãÊï∞, Ëã•Êó†Ê≥ïËé∑ÂèñÂàôÈªòËÆ§‰∏∫4
  static std::size_t default_thread_count()
  {
    auto n = std::thread::hardware_concurrency();
    return n == 0 ? 4 : n;
  }

  /// @brief È™åËØÅÁ∫øÁ®ãÊï∞ÊòØÂê¶ÂêàÊ≥ï, 1 <= count <= 4096
  static std::size_t validate_thread_count(std::size_t count)
  {
    if (count < 1 || count > 4096) throw std::invalid_argument("invalid thread_count: must be in range [1, 1024]");
    return count;
  }

  /// @brief ÂêØÂä®Á∫øÁ®ãÊ±†, ÂàõÂª∫ÊåáÂÆöÊï∞ÈáèÁöÑÂ∑•‰ΩúÁ∫øÁ®ã
  /// @param thread_count Á∫øÁ®ãÊ±†‰∏≠Á∫øÁ®ãÁöÑÊï∞Èáè
  void launch_threads(std::size_t thread_count)
  {
    if (!workers_.empty()) return;  // Â∑≤ÁªèÂàùÂßãÂåñËøá
    for (std::size_t i = 0; i < thread_count; ++i)
    {
      // ÂàõÂª∫Âπ∂ÂêØÂä®Â∑•‰ΩúÁ∫øÁ®ã
      workers_.emplace_back([this] {
        while (true)
        {
          task_t task;
          {
            std::unique_lock<std::mutex> lock(mtx_);
            // Á≠âÂæÖÁõ¥Âà∞‰ªªÂä°ÈòüÂàó‰∏≠Êúâ‰ªªÂä°, ÊàñËÄÖÁ∫øÁ®ãÊ±†Â∑≤ÂÅúÊ≠¢
            cv_.wait(lock, [this] { return !running_ || !task_queue_.empty(); });
            if (!running_ && task_queue_.empty()) return;  // Â¶ÇÊûúÁ∫øÁ®ãÊ±†Â∑≤ÁªèÂÅúÊ≠¢Âπ∂‰∏îÈòüÂàó‰∏∫Á©∫, ÈÄÄÂá∫Á∫øÁ®ã
            task = std::move(task_queue_.front());         // ‰ªéÈòüÂàó‰∏≠ÂèñÂá∫‰ªªÂä°
            task_queue_.pop();
          }
          ++busy_count_;
          task();  // ÊâßË°å‰ªªÂä°
          --busy_count_;
          // Âà§Êñ≠‰ªªÂä°ÊòØÂê¶Â∑≤ÂÖ®ÈÉ®ÂÆåÊàê
          if (busy_count_ == 0 && task_queue_.empty())
          {
            std::lock_guard<std::mutex> lock(mtx_done_);
            if (busy_count_ == 0 && pending_tasks() == 0)  // ‰∫åÊ¨°Á°ÆËÆ§, ÈÅøÂÖçÁ´ûÊÄÅ
            {
              cv_done_.notify_all();
            }
          }
        }
      });
    }
  }

 private:
  std::vector<std::thread> workers_;  // Â∑•‰ΩúÁ∫øÁ®ãÈõÜÂêàÔºåÁî®‰∫éÂπ∂ÂèëÊâßË°å‰ªªÂä°
  std::queue<task_t> task_queue_;     // Á≠âÂæÖÊâßË°åÁöÑ‰ªªÂä°ÈòüÂàó
  std::condition_variable cv_;        // Êù°‰ª∂ÂèòÈáèÔºåÁî®‰∫éÈÄöÁü•Â∑•‰ΩúÁ∫øÁ®ãÊúâÊñ∞‰ªªÂä°Âà∞Êù•
  mutable std::mutex mtx_;            // ‰∏ª‰∫íÊñ•ÈîÅÔºå‰øùÊä§‰ªªÂä°ÈòüÂàóÂíå‰∏éÂÖ∂Áõ∏ÂÖ≥ÁöÑÁä∂ÊÄÅ

  std::atomic<std::size_t> busy_count_{0};  // Ê≠£Âú®ÊâßË°å‰ªªÂä°ÁöÑÁ∫øÁ®ãÊï∞Èáè
  std::atomic<bool> running_{true};         // Á∫øÁ®ãÊ±†ÊòØÂê¶Â§Ñ‰∫éËøêË°åÁä∂ÊÄÅ

  mutable std::mutex mtx_done_;      // Áî®‰∫é‰øùÊä§ÂÆåÊàêÈÄöÁü•ÁöÑ‰∫íÊñ•ÈîÅ(wait_all Áî®)
  std::condition_variable cv_done_;  // Êù°‰ª∂ÂèòÈáèÔºåÁî®‰∫éÁ≠âÂæÖÊâÄÊúâ‰ªªÂä°ÊâßË°åÂÆåÊØï(ÈÖçÂêà wait_all ‰ΩøÁî®)
};
